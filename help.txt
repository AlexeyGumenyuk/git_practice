git init - инициализировать локальный репозитори
git remote add <name (обычно origin)> <ssh репозитория (кнопка code на его странице)> - связать локальный и удалённый репозитории
git remote -v - проверить привязан ли локальный репозиторий к удалённому
git add <имя или флаг --all> - подготовить файлы к сохранению
git status - проверить состояние локального репозитория
git commit -m "Комментарий" - выполнить коммит
git log - просмотреть историю коммитов
git push - отправить изменения на удалённый репозиторий
git pull - получить изменения на локальный репозиторий
ssh-keygen -t rsa -b 4096 -C <эл. почта, к которой привязан аккаунт на GitHub> - генерация пары SSH ключей (один раз для аккаунта)
